    title Консольный мессенджер - контейнеры

    Container(server, "Сервер", "C++ программа", "Принимает подключения, пересылает сообщения между клиентами")

    Rel(client, server, "отправляет/получает сообщения", "TCP/IP")
    Rel(server, db, "сохраняет и загружает данные", "SQL")
